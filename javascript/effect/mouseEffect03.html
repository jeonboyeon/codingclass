const cursor = document.querySelector(".mouse__cursor"); // 커서
// const circleW = cursor.offsetWidth; // 너비 200 | border 값 포함 ↓
// const circleH = cursor.offsetHeight; ↓
const circle = cursor.getBoundingClientRect(); // ↓

// getBoundingClientRect 썼을 때 circle 배열 (요소를 마우스 중앙에 배치하기 위해 각 값들을 구함)
// console.log(circle);
// const DOMRect = {
//     x: 0,
//     y: 0,
//     bottom: 200,
//     left: 0,
//     right: 200,
//     top: 0,
//     width: 200,
// };

// const circle2 = cursor.clientWidth; // 너비 190 | border 값 미포함

window.addEventListener("mousemove", (e) => {
    // 마우스를 움직일 때
    // 배열로 구했던 whidth와 height 값을 불러와 pageX 와 pageY에 나눠준다.
    gsap.to(cursor, { duration: 0.5, left: e.pageX - circle.width / 2, top: e.pageY - circle.height / 2 });
});
