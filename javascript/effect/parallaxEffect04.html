// 재귀 함수 (자기 자신을 다시 호출)
function scroll() {
    let scrollTop = window.pageYOffset || document.documentElement.scrollTop || window.screenTop;

    document.querySelectorAll(".content__item").forEach((item) => {
        if (scrollTop > item.offsetTop - window.innerHeight / 1.3) { // 좀 더 빨리 show를 붙이기 위해
            item.classList.add("show");
        }
    });

    requestAnimationFrame(scroll); // 브라우저에 최적화 되어 있어 움직임이 부드럽고 부담이 덜 하다. (1분에 60번)
}
