//선택자

const searchBox = document.querySelector(".search__box input"); // HTML 검색 박스
const searchList = document.querySelectorAll(".search__list ul li"); // 태그 목록 리스트
const searchInfo = document.querySelector(".search__info .num");    //css 속성 갯수

//전체 갯수 구하기
searchInfo.textContent = searchList.length;

//searchBox
searchBox.addEventListener("keyup", () => {
    const searchWord = searchBox.value;

    searchList.forEach(el => {
        const cssName = el.dataset.name; //css 속성 모든 값이 들어가 있다.
        const cssType = el.dataset.type; //css 속성 모든 값이 들어가 있다.
        console.log(cssName);

        "all".includes("all"); //true

        if(cssName.includes(searchWord)){
            el.classList.add("show");
        } else {
            el.classList.remove("show");
        }
    });
});
