    <script>
        //선택자

        const searchBox = document.querySelector(".search__box input"); // HTML 검색 박스
        const searchList = document.querySelectorAll(".search__list ul li"); // 태그 목록 리스트

        searchBox.addEventListener("keyup", () => {
            const searchWord = searchBox.value; // 사용자가 입력한 키워드
            // console.log(searchWord);

            searchList.forEach(el => {
                const cssName = el.dataset.name;

                if (cssName.indexOf(searchWord)) {
                    el.classList.add("hide"); // hide 속성을 추가한다.
                } else {
                    // Ex) abbr.indexOf("abbr") 은 0
                    el.classList.remove("hide"); // hide를 지워 보여준다. (indexOf 값이 0이 되면 0이 되는 값만 hide 속성을 없앤다.)
                }
            });
        });
    </script>
