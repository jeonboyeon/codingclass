    function solution(arr, divisor) {
        var answer = [];

        const div = arr.filter((el) => el % divisor == 0);
        answer = div.length > 0 ? div.sort((a, b) => b - a) : [-1];

        return answer;
    }
    document.write(solution([1, 0, 5, 3, 7, 11, 8, 10, 15], 3));
}
